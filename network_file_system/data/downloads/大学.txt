import requests
from tqdm import tqdm
import os

def download_with_tqdm(url, save_path):
    """使用 tqdm 下载文件"""
    response = requests.get(url, stream=True)
    total_size = int(response.headers.get('content-length', 0))
    
    # 创建进度条
    progress_bar = tqdm(total=total_size, unit='B', unit_scale=True, desc='下载中')
    
    with open(save_path, 'wb') as file:
        for chunk in response.iter_content(chunk_size=8192):
            if chunk:
                file.write(chunk)
                progress_bar.update(len(chunk))
    
    progress_bar.close()
    print(f"下载完成: {save_path}")

def upload_with_tqdm(file_path, upload_url):
    """使用 tqdm 上传文件"""
    total_size = os.path.getsize(file_path)
    
    # 创建进度条
    progress_bar = tqdm(total=total_size, unit='B', unit_scale=True, desc='上传中')
    
    def callback(bytes_read):
        progress_bar.update(bytes_read - progress_bar.n)
    
    # 读取文件并上传
    with open(file_path, 'rb') as file:
        files = {'file': (os.path.basename(file_path), file)}
        response = requests.post(upload_url, files=files)
    
    progress_bar.close()
    
    if response.status_code == 200:
        print("上传成功")
    else:
        print(f"上传失败: {response.status_code}")
